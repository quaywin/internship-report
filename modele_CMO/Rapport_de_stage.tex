	\hbox to \hsize{%
	\rlap{\vtop{\raggedright #1}}%

	\hbox to \hsize{%

\def\maketitle{%
	\thispagestyle{empty}\vbox to \vsize{%
		\haut{\@logoUniversity}{}{\@logoLabo}
		\vspace{.7cm}
		\haut{}{\@blurb}{}
		\vfill
		\vspace{.5cm}
		\begin{flushleft}
		\usefont{OT1}{ptm}{m}{n}
		\huge \@title
		\end{flushleft}
		\par
		\hrule height 4pt
		\par
		\begin{flushright}
		\usefont{OT1}{phv}{m}{n}
		\Large \@author
		\par
		\end{flushright}
		\vspace{1cm}
		\haut{}{\@logoCentral}{}
		\vfill
		\vfill
		\bas{}{\@location, le \@date}{}}%
	\cleardoublepage
}
